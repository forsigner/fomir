```jsx
import { useFieldState } from 'fomir-react'
```

```jsx live
function BasicForm() {
  const NameValue = () => {
    const { value } = useFieldState('userName')
    return <div>Name: {value}</div>
  }

  const form = useForm({
    onSubmit(values) {
      alert(JSON.stringify(values, null, 2))
      console.log('values', values)
    },
    // register component
    components: {
      NameValue,
    },

    children: [
      {
        component: 'NameValue',
      },
      {
        name: 'userName',
        component: 'Input',
        value: 'Bill',
      },
      {
        component: 'Submit',
        text: 'submit',
      },
    ],
  })

  return <Form form={form} />
}
```
