```jsx live=true
function BasicForm() {
  const form = useForm({
    onSubmit(values) {
      alert(JSON.stringify(values, null, 2))
      console.log('values', values)
    },

    children: [
      {
        label: 'First Name',
        name: 'firstName',
        component: 'Input',
        validators: {
          required: 'First name is required',
        },
        value: '',
      },
      {
        label: 'Last Name',
        name: 'lastName',
        component: 'Input',
        validators: {
          required: 'Last name is required',
          minLength: [6, 'min length is 6'],
        },
        value: '',
      },
      {
        component: 'Submit',
        text: 'submit',
      },
    ],
  })

  return <Form form={form} />
}
```
